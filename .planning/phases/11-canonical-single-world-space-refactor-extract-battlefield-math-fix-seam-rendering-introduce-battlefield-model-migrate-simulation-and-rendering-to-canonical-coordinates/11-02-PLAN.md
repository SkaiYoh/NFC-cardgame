static void game_draw_entities_for_viewport(GameState *g, const Player *viewportPlayer) {
    for (int pid = 0; pid < 2; pid++) {
        const Player *owner = &g->players[pid];
        const Player *opponent = &g->players[1 - pid];
        for (int i = 0; i < owner->entityCount; i++) {
            const Entity *e = owner->entities[i];
            if (viewportPlayer == owner) {
                entity_draw(e);
            } else if (e->position.y < owner->playArea.y) {
                Vector2 mappedPos = game_map_crossed_world_point(owner, opponent, e->position);
                AnimState crossed = e->anim;
                game_apply_crossed_direction(e, owner, opponent, &crossed);
                sprite_draw(e->sprite, &crossed, mappedPos, e->spriteScale);
            }
        }
    }
}
```

<!-- Key constants -->
SPRITE_FRAME_SIZE = 79  (from config.h)
Entity.spriteScale = 2.0f (default from entities.c:31)
Effective sprite height = SPRITE_FRAME_SIZE * spriteScale = 79 * 2.0 = 158px

<!-- The seam bug -->
Current logic: entity drawn in opponent viewport ONLY when e->position.y < owner->playArea.y
Problem: entity at e->position.y = 10 with sprite height 158px means sprite extends from y=-69 to y=89.
         The owner viewport clips at y=0 (playArea.y), so the bottom 10px of the sprite are clipped.
         The opponent viewport only draws it when y < 0, missing the 0..10 range.
Fix: draw entity in BOTH viewports when sprite bounds overlap the seam (playArea.y boundary).

    for (int i = 0; i < owner->entityCount; i++) {
        const Entity *e = owner->entities[i];
        // Compute sprite half-height for seam overlap detection
        float spriteHalfH = (SPRITE_FRAME_SIZE * e->spriteScale) * 0.5f;
        // The seam boundary is at the top edge of the owner's play area
        float seamBorder = owner->playArea.y;

        if (viewportPlayer == owner) {
            // Owner viewport: always draw owner's entities
            entity_draw(e);

            // SEAM FIX: If entity is near the seam and its sprite overlaps into
            // opponent territory, ALSO draw it in the owner viewport using the
            // opponent's mapped position. This handles the case where the entity
            // is just barely past the border but its sprite bottom is still visible.
            // (This case is already handled by the opponent viewport branch below.)
        } else {
            // Opponent viewport: draw if entity has crossed OR if sprite overlaps seam
            bool crossed = (e->position.y < seamBorder);
            bool spriteOverlapsSeam = (e->position.y - spriteHalfH < seamBorder) &&
                                      (e->position.y + spriteHalfH > seamBorder);

            if (crossed || spriteOverlapsSeam) {
                Vector2 mappedPos = game_map_crossed_world_point(owner, opponent, e->position);
                AnimState crossedAnim = e->anim;
                game_apply_crossed_direction(e, owner, opponent, &crossedAnim);
                sprite_draw(e->sprite, &crossedAnim, mappedPos, e->spriteScale);
            }
        }
    }
    ```

    The key change: the condition `e->position.y < owner->playArea.y` is expanded to also check `spriteOverlapsSeam`. This draws the sprite in the opponent viewport when the sprite bounds straddle the seam, even if the entity center hasn't fully crossed yet. The existing `game_map_crossed_world_point` and `game_apply_crossed_direction` functions handle the position and direction mapping correctly.

    Keep the existing `game_map_crossed_world_point` and `game_apply_crossed_direction` functions unchanged -- they work correctly and will be deleted in Plan 05.
